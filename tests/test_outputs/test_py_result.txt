import os
import flet as ft
import asyncio

from src.LoginView import LoginView
from src.MainView import MainView

class FletUI:
 def __init__(self, core):
 self.page: ft.Page | None = None
 self.core = core

 #=========WEB===================

 def show_login_web(self):
 if self.page is None:
 return
 
 self.page.views.clear()

 login_view = LoginView(
 core=self.core, 
 on_login=self.show_main_web
 )
 self.page.views.append(login_view)
 self.page.active_view = login_view
 self.page.update()
 
 def show_main_web(self, user):
 print("Открываем главное окно")
 if self.page is None:
 return
 
 self.page.views.clear()

 main_view = MainView(
 on_logout=self.show_login_web
 )
 self.page.views.append(main_view)
 self.page.update()
 self.page.active_view = main_view
 main_view.start(core=self.core, current_user=user)

 
 def main_web(self, page: ft.Page):
 self.page = page
 page.theme_mode = "light"
 self.show_login_web()

 def run_web(self):
 os.environ["FLET_SECRET_KEY"] = "my-secret-key-24022026"
 ft.app(
 self.main_web,
 view=ft.AppView.WEB_BROWSER,
 upload_dir=self.core.config.UPLOAD_DIR,
 port=8550, 
 host="0.0.0.0" # важно для Docker!

 )

 # ========DESKTOP============
 
 def show_login_desktop(self):
 if self.page is None:
 return

 login_view = LoginView(
 core=self.core, 
 on_login=self.show_main_desktop
 )
 self.page.views.clear()
 self.page.views.append(login_view)
 self.page.active_view = login_view

 self.page.window.resizable = True
 self.page.window.min_width = 400
 self.page.window.max_width = 400
 self.page.window.min_height = 400
 self.page.window.max_height = 400
 #self.page.window.width = 400
 #self.page.window.height = 400
 #self.page.window.resizable = False
 asyncio.create_task(self.page.window.center())
 self.page.update()

 

 

 def show_main_desktop(self, user):
 if self.page is None:
 return
 
 self.page.views.clear()

 main_view = MainView(
 on_logout=self.show_login_desktop
 )
 self.page.views.append(main_view)
 
 self.page.window.resizable = True
 self.page.window.min_width = 1300
 self.page.window.max_width = 1300
 self.page.window.min_height = 700
 self.page.window.max_height = 700
 #self.page.window.width = 1300
 #self.page.window.height = 700
 #self.page.window.resizable = False
 asyncio.create_task(self.page.window.center())

 self.page.update() 

 self.page.active_view = main_view
 main_view.start(core=self.core, current_user=user)

 def main_desktop(self, page: ft.Page):
 self.page = page
 page.theme_mode = "light"
 page.window_frameless = True
 self.show_login_desktop()

 def run_desktop(self):
 ft.app(
 self.main_desktop,
 view=ft.AppView.FLET_APP,
 upload_dir=self.core.config.UPLOAD_DIR
 )
